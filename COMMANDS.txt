# На своём Windows (в папке проекта):
cd C:\Users\makc0\Desktop\SongForge

# Посмотреть коммиты:
git log --oneline -10

# Откатить index.html к версии ДО поломки (замени ABC123 на коммит из лога выше):
git checkout ABC123 -- index.html

# Закоммитить и запушить:
git add index.html
git commit -m "revert to working"
git push origin main

# На сервере:
ssh root@195.19.20.245
(введи пароль)

# Скачать откатанную версию:
curl -fsSL "https://raw.githubusercontent.com/maxxx010182/SongForge/main/index.html" -o ~/SongForge/index.html

# Если TLS — этот:
curl -fsSL "https://codeload.github.com/maxxx010182/SongForge/tar.gz/main" -o /tmp/sf.tar.gz
tar -xzf /tmp/sf.tar.gz -C /tmp --strip-components=1 SongForge-main/index.html
cp /tmp/index.html ~/SongForge/index.html
rm -f /tmp/sf.tar.gz /tmp/index.html

pm2 restart songforge
